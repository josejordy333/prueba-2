y++) { terrain[x][y] = []; for (let z = 0; z < size; z++) { if (y < height / 2) { terrain[x][y][z] = 1; // Tierra } else { terrain[x][y][z] = 0; // Aire } } } } return terrain; } import { generateTerrain } from './terrain.js'; renderer = new THREE.WebGLRenderer({ antialias: true }); renderer.setSize(window.innerWidth, window.innerHeight); document.body.appendChild(renderer.domElement); // Luz const ambientLight = new THREE.AmbientLight(0xffffff); x++) { for (let y = 0; y < terrainHeight; y++) { for (let z = 0; z < terrainSize; z++) { if (terrainData[x][y][z] === 1) { const block = new THREE.Mesh(blockGeometry, blockMaterial); block.position.set(x - terrainSize / 2, y, z - terrainSize / 2); scene.add(block); } } } } // Cámara camera.position.set(0, 10, 30); camera.lookAt(0, 0, 0); // Controles const controls = new THREE.PointerLockControls(camera, document.body); document.addEventListener('click', () => controls.lock()); break; case 'ArrowRight': case 'KeyD': controls.moveRight(0.1); break; } } document.addEventListener('keydown', onKeyDown); // Animación function animate() { requestAnimationFrame(animate); renderer.render(scene, camera);
